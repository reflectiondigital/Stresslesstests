This module is intended as a generic form building API, allowingmodule developers to include forms to be created byusers/administrators.(it's also pretty heavily still in progress).Basic usage is as follows:<?php$form = forms_load($form_id);if ($_POST['op']) {  $error = forms_validate($form, $_POST['edit']);  if ($error) {    // HANDLE THE ERROR HERE    print "ERROR";  }  else {    // DO SOMETHING WITH $_POST['edit'] HERE.    print_r($_POST['edit']);  }}// put any pre-form things here.$content = forms_render($form, $_POST['edit'], $error);$content.= form_submit(t("Submit"));print form($content);?>
